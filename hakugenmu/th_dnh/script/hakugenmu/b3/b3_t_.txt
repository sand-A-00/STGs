




script_enemy_main {
let teki_png = "script\hakugenmu\img\sakuya.png";
let teki_png_i = "script\hakugenmu\img\sakuya_.png";
let tatie =teki_png;
let dfx=0;
let dfy=0;
    @Initialize {
CreateEventFromScript("b3_event");

//setx , sety は初期の右上の位置。 そしてsetMoveP....は中央への移動。
        SetX(GetClipMinX);
        SetY(GetClipMinY - 64);
         SetInvincibility(10);
        SetLife(1000);
        LoadGraphic(teki_png);
        LoadGraphic(teki_png_i);
        SetTexture(tatie);
        SetGraphicRect(0, 0, 63, 63);
        SetTexture(teki_png);
dfx=GetX;
dfy=GetY;
a;

    }

    @MainLoop {
        SetCollisionA(GetX, GetY, 24);
        SetCollisionB(GetX, GetY, 24);



if(OnEvent()==false){
//VanishEnemy();
AddLife(-1000);

}

        




       yield;
       
    }

    @DrawLoop {
if(GetX!=dfx || GetY!=dfy){
	tatie = teki_png_i;
	if(GetX<dfx){SetGraphicScale(1.3,1.3);}
	else{SetGraphicScale(-1.3,1.3);}
}
else{
tatie = teki_png;
SetGraphicScale(1.3,1.3);
}
dfx=GetX;
dfy=GetY;
        DrawGraphic(GetX, GetY);
        SetGraphicRect(0, 0, 63, 63);
        SetTexture(tatie);

   }

        



    @Finalize {

    }


task a{
while(GetEventStep==0){yield;}
SetMovePosition02(GetCenterX,GetClipMinY+ 128,60);
}

}
