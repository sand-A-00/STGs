




script_enemy_main {
let teki_png = "script\hakugenmu\img\rumia.png";
let teki_png_i = "script\hakugenmu\img\rumia_.png";
let tatie =teki_png;
let dfx=0;
let dfy=0;
    @Initialize {
CreateEventFromScript("b1_event");

//setx , sety は初期の右上の位置。 そしてsetMoveP....は中央への移動。
        SetX(GetClipMinX + 32);
        SetY(GetClipMinY + 32);
        SetMovePosition02(GetCenterX, GetClipMinY + 128, 30);
        SetLife(100);
        LoadGraphic(teki_png);
        LoadGraphic(teki_png_i);
        SetTexture(tatie);
        SetGraphicRect(0, 0, 63, 63);
        SetTexture(teki_png);


    }

    @MainLoop {
        SetCollisionA(GetX, GetY, 24);
        SetCollisionB(GetX, GetY, 24);



if(OnEvent()==false){
AddLife(-100);



}

        




       yield;
       
    }

    @DrawLoop {
if(GetX!=dfx || GetY!=dfy){
	tatie = teki_png_i;
	if(GetX<dfx){SetGraphicScale(1.3,1.3);}
	else{SetGraphicScale(-1.3,1.3);}
}
else{
tatie = teki_png;
SetGraphicScale(1.3,1.3);
}
dfx=GetX;
dfy=GetY;
        DrawGraphic(GetX, GetY);
        SetGraphicRect(0, 0, 63, 63);
        SetTexture(tatie);

   }


        



    @Finalize {

    }


}
