	script_event t3_text_end
	{
		@Initialize
		{
			LoadGraphic("script\hakugenmu\img\reimu.png");

                                                
		}
		
		@MainLoop
		{

			SetChar(LEFT,"script\hakugenmu\img\reimu.png");
			SetGraphicRect(LEFT,1,1,128,256);

			MoveChar(LEFT,FRONT);//左側のキャラクタを前に移動
			MoveChar(RIGHT,BACK);//右側のキャラクタを後に移動
			TextOut("\c[BLUE]門番てのは、客人を館に案内するものじゃないのかしら。");//文字描画
			Wait(60);//30フレーム待つ
			End();//最後にEnd()は必須
		}
		
		@Finalize
		{
			DeleteGraphic("script\hakugenmu\img\reimu.png");
		}
	}
