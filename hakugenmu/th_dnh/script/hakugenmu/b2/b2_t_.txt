    @Initialize {
CreateEventFromScript("b2_event");

//setx , sety は初期の右上の位置。 そしてsetMoveP....は中央への移動。
        SetX(GetCenterX);
        SetY(GetClipMinY + 128);
         SetInvincibility(10);
        SetLife(100);
        LoadGraphic(teki_png);
        LoadGraphic(teki_png_i);
        SetTexture(teki_png);
        SetGraphicRect(0, 0, 63, 63);
        SetTexture(teki_png);
SummonGraphic;

    }

    @MainLoop {
        SetCollisionA(GetX, GetY, 24);
        SetCollisionB(GetX, GetY, 24);

AddLife(-100);


}

        




       yield;
       
    }

    @DrawLoop {

      SetTexture(teki_png);
        SetGraphicRect(0, 0, 63, 63);
        DrawGraphic(GetX, GetY);

   }

        



    @Finalize {

    }


task SummonGraphic{

warp = true;

loop(10){
SetGraphicScale(warpX,warpY);

warpY -= 0.1;
warpX += 0.05;
Wait(2);

}

}
			function Wait(let frame)
			{
				loop(frame){yield;}
			}
